    <!DOCTYPE html>

    <script>
    window.onload = function () {
        var box = document.querySelector(".showBox");
        box.ondragover = function (e) {
            e.preventDefault();
        };
        var imgs = document.querySelectorAll("img");
        var divs = document.querySelectorAll(".colorBox");
        var move = null;
        for (var i = 0; i < imgs.length; i++) {
            var obj = imgs[i];
            obj.ondragstart = function () {
                move = this;
            }
        }
        for (var i = 0; i < divs.length; i++) {
            var obj = divs[i];
            obj.ondragstart = function () {
                move = this;
            };
        }
        box.ondrop = function () {
            if (move.src) {
                this.style.background = "url('" + move.src + "') no-repeat center/cover";
            } else {
                var color = window.getComputedStyle(move).backgroundColor;
                this.style.backgroundColor = color;
            }
        };
    };
    </script>

    <!DOCTYPE html>
    <html>
    <head>
    <meta charset="UTF-8">
    <title></title>
    <style type="text/css">
        html, body {
            height: 100%;
        }

        body {
            /*body元素 默认就是一个盒子 是没有高度的*/
            border: 10px dashed gray;
            margin: 0;
        }
    </style>
    <script>
        window.onload = function () {
            document.body.ondragover = function (e) {
                e.preventDefault();
            };
            document.body.ondrop = function (e) {
                e.preventDefault();
                console.log(e);
                var file = e.dataTransfer.files[0];
                var reader = new FileReader();
                reader.readAsDataURL(file);
                reader.onload = function () {
                    document.body.style.background = "url('" + reader.result + "') no-repeat center/cover";
                };
            };
        };
       </script>
    </head>
    <body>
    </body>
    </html>
